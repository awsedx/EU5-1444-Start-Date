	allow = {
		leader_country ?= { tribal_cohesion > 10 }
	}
	locked = {
		custom_tooltip = {
			text = 5_years_cd

				has_variable = local_adm_law_cooldown_variable
			}
		}

	tribal_administration_policy = {
		wants_this_policy_bias = {
			add = {
				desc = "POLICY_BIAS_GOOD_IDEA"
				value = 20
			}
		}

		on_activate = {
			leader_country = { 
				set_variable = {
					name = local_adm_law_cooldown_variable
					value = yes
					years = 5
				}
				add_tribal_cohesion = tribal_cohesion_weak_penalty
				if = {
					limit = {
						ruler_or_regent ?= { has_variable = difficult_sinicization_variable }
					}
					add_tribal_cohesion = tribal_cohesion_mild_penalty
					add_stability = stability_mild_penalty

			}
			every_international_organization_member = {
				change_societal_value = {

				}
			}
		}
		years = 5
		country_modifier = {
			monthly_towards_unsinicized = societal_value_monthly_move

khalkha_unification_law = {
    law_category = centralization
    type = international_organization

    potential = {
        international_organization_type = khalkha_confederation
    }

    allow = {
        international_organization = {
            total_members >= 2
        }
    }

    custom_tags = { forbids_no_policy }

    no_unification_policy = {
        wants_propose_policy = {
            subtract = { value = 1000 }
        }
    }

    unify_khalkha_policy = {
        wants_propose_policy = {
            add = { value = 500 }
        }

        wants_this_policy_bias = {
            add = {
                value = prev_opinion_of_this
                multiply = 0.3
            }
            add = {
                value = root.relative_strength(scope:recipient)
                multiply = 50
            }
        }

        on_activate = {
            custom_tooltip = khalkha_unification_tt

            root.leader_country = {
                form_country = KHL
                if = {
			        limit = { country_rank_level < 3 }
			        set_country_rank_effect = { rank = country_rank:rank_kingdom }
		        }            }

            every_international_organization_member = {
                limit = { NOT = { this = root.leader_country } }
                annex = root.leader_country
            }

            disband_international_organization = yes
        }
    }
}
