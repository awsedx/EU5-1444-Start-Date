namespace = australia_pops

australia_pops.1 = {
	type = country_event
	title = australia_pops.1.title
	desc = australia_pops.1.desc
	
	trigger = {
		AND = {
			current_year > 1400
			OR = {
				tag = WUW
				tag = EOA
				tag = TIW
				tag = CKC
			}
		}
	}
	
	fire_only_once = yes
	
	immediate = {
		capital = {
			save_scope_as = target_capital
		}
	}
	
	option = {
		name = australia_pops.1.a
		
		ai_chance = { factor = 0 }
		
		capital = {
			create_pop = {
				culture = owner.culture
				religion = owner.religion
				type = pop_type:peasants
				size = 5000
			}
		}
	}
	
	option = {
		name = australia_pops.1.b
		
		ai_chance = { factor = 100 }
		
		# Refuse - do nothing
	}
}
